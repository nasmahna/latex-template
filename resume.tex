% 3. Custom Indenting Environment (Requires changepage package)
\newenvironment{subs}
  {\begin{adjustwidth}{2em}{0pt} \footnotesize} % <-- CHANGED: Added \footnotesize here
  {\end{adjustwidth}}

% Command for Subheadings (Job Title, Degree, etc.)
% Original: \newcommand{\subheading}[2]{% \textbf{#1} \hfill \emph{#2} \\ }
\newcommand{\subheading}[2]{%
    \footnotesize\textbf{#1} \hfill \emph{#2} \\ % <-- CHANGED: Used \footnotesize
}

% Command for Job/Project Details (Making top line \footnotesize)
\newcommand{\jobdetails}[3]{%
    \footnotesize \textbf{#1} \hfill #2 \\ % Role / project and secondary detail (e.g. location or tech)
    \emph{\small #3} \\ % Date range or link / extra detail
}
